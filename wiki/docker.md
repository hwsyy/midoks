# docker 常用命令
```
docker info
docker version
docker-compose --version
docker-machine --version

docker run -d -p 80:80 --name webserver nginx
docker ps


docker start web 			#启动容器名为web的容器
docker exec -it web bash 	#进入容器
docker stop web 			#停止容器
docker rm web 				#删除容器


```
